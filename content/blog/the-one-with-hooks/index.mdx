## React.createClass()

In the event that you've been around the React game long enough, you'll recollect the React.createClass API. It was the first manner by which we'd make React segments. The entirety of the data you'd use to depict the segment would be passed as an item to createClass.

```jsx
const ReposGrid = React.createClass({
  getInitialState() {
    return {
      repos: [],
      loading: true,
    };
  },
  componentDidMount() {
    this.updateRepos(this.props.id);
  },
  componentDidUpdate(prevProps) {
    if (prevProps.id !== this.props.id) {
      this.updateRepos(this.props.id);
    }
  },
  updateRepos(id) {
    this.setState({ loading: true });

    fetchRepos(id).then((repos) =>
      this.setState({
        repos,
        loading: false,
      })
    );
  },
  render() {
    const { loading, repos } = this.state;

    if (loading === true) {
      return <Loading />;
    }

    return (
      <ul>
        {repos.map(({ name, handle, stars, url }) => (
          <li key={name}>
            <ul>
              <li>
                <a href={url}>{name}</a>
              </li>
              <li>@{handle}</li>
              <li>{stars} stars</li>
            </ul>
          </li>
        ))}
      </ul>
    );
  },
});
```
